|--------|---------|--------|

### AI Enhancements
- **Smart Due Dates**: AI suggests due dates based on task type and workload
- **Auto-Categorization**: AI assigns category from task title
- **Reminder Timing**: AI suggests optimal reminder time based on user patterns
- **Related Linking**: AI auto-links tasks to jobs/invoices when mentioned

## Database Schema

```sql
CREATE TABLE tasks (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  company_id UUID NOT NULL REFERENCES companies(id),
  created_by UUID NOT NULL REFERENCES users(id),
  assigned_to UUID REFERENCES users(id),

  -- Content
  title TEXT NOT NULL,
  description TEXT,
  category TEXT DEFAULT 'general',

  -- Relationships
  job_id UUID REFERENCES jobs(id),
  client_id UUID REFERENCES clients(id),
  vendor_id UUID REFERENCES vendors(id),
  invoice_id UUID REFERENCES invoices(id),
  related_entity_type TEXT,
  related_entity_id UUID,

  -- Status
  status TEXT DEFAULT 'backlog',
  priority TEXT DEFAULT 'medium',

  -- Dates
  due_date TIMESTAMP WITH TIME ZONE,
  reminder_date TIMESTAMP WITH TIME ZONE,
  completed_at TIMESTAMP WITH TIME ZONE,

  -- Meta
  notes TEXT,
  ai_generated BOOLEAN DEFAULT false,
  ai_source TEXT,

  created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
  updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- Indexes
CREATE INDEX idx_tasks_company_id ON tasks(company_id);
CREATE INDEX idx_tasks_assigned_to ON tasks(assigned_to);
CREATE INDEX idx_tasks_job_id ON tasks(job_id);
CREATE INDEX idx_tasks_status ON tasks(status);
CREATE INDEX idx_tasks_due_date ON tasks(due_date);
CREATE INDEX idx_tasks_priority ON tasks(priority);

-- RLS Policy
ALTER TABLE tasks ENABLE ROW LEVEL SECURITY;
CREATE POLICY tasks_company_isolation ON tasks
  USING (company_id = current_setting('app.company_id')::UUID);
```
